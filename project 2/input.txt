bool isPrime(int num) {
    if (num <= 1) {
        return false;
    }
    for (int i = 2; i * i <= num; i++) {
        if (num % i == 0) {
            return false;
        }
    }
    return true;
}

int main() {
    int number = 9;
    int result = isPrime(number);

    if (result) {
        printf("Prime number.\n");
    } else {
        printf("Not a prime number.\n");
    }

    return 0;
}
